The one feature which was the hugest headache is this:

// @errors: 2345
const message = 'Hello world!';
//    ^?
const sum = (a: number, b: number) => a + b;
console.log(sum(1, message));

This is achieved by a little tool called Twoslash, used in the TypeScript website itself. You can do more cool stuff:

// Hello
console.error("This is an error")
// @error: This is an error
console.log("This is a log")
// @log: This is a log
console.warn("This is a warning")
// @warn: This is a warning

// @noErrors
useless.code();
more.useless.code();
some.more();
const date = new Date();
//           ^^^^^^^^^^
console.log(date)




// @annotate: right - Creates date. Some more blah blah to stress test this
const date = new Date();

interface Person {
	name: string;
	age: number;
}

// ---cut---

const a: Person = {
	name: 'John',
	age: 42,
};

function increment() {
	counter++;
	document.querySelector('p.inc').innerHTML = `The number is ${counter}`;
}
</script>
<button onclick='increment()'>Click me</button>
<p class='inc'>The number is 1</p>
